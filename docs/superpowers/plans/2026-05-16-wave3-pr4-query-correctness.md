Create `tests/property_test.rs`:

```rust
//! Cross-feature property tests (#221).
//!
//! Compares Minigraf query results against a deliberately simple reference
//! evaluator for randomly-generated small graphs and queries.
//!
//! Run: cargo test --test property_test
//! Increase cases: PROPTEST_CASES=1000 cargo test --test property_test

use minigraf::db::Minigraf;
use minigraf::QueryResult;
use proptest::prelude::*;

// ── Types used in this file ───────────────────────────────────────────────────

/// A minimal in-memory fact for the reference evaluator.
#[derive(Debug, Clone)]
struct TestFact {
    entity: usize,     // small integer entity ID
    attribute: String, // e.g. ":color"
    value: TestValue,
}

#[derive(Debug, Clone, PartialEq)]
enum TestValue {
    Str(String),
    Int(i64),
    Bool(bool),
}

impl TestValue {
    fn to_edn(&self) -> String {
        match self {
            TestValue::Str(s) => format!(r#""{s}""#),
            TestValue::Int(n) => n.to_string(),
            TestValue::Bool(b) => b.to_string(),
        }
    }
}

/// A simple query pattern: find entities that have a specific attribute.
#[derive(Debug, Clone)]
struct TestQuery {
    /// The attribute to match on.
    attribute: String,
    /// If Some, also require a specific value.
    value_filter: Option<TestValue>,
    /// If true, exclude entities that ALSO have `negation_attr`.
    negation_attr: Option<String>,
}

// ── Reference evaluator (naive, independent of production code) ───────────────

fn ref_eval(facts: &[TestFact], query: &TestQuery) -> Vec<usize> {
    // Find all entities with the matching attribute (and optional value).
    let mut matched: Vec<usize> = facts
        .iter()
        .filter(|f| {
            f.attribute == query.attribute
                && match &query.value_filter {
                    None => true,
                    Some(v) => &f.value == v,
                }
        })
        .map(|f| f.entity)
        .collect();

    // Apply negation: remove entities that also have the negation attribute.
    if let Some(neg_attr) = &query.negation_attr {
        let neg_entities: std::collections::HashSet<usize> = facts
            .iter()
            .filter(|f| &f.attribute == neg_attr)
            .map(|f| f.entity)
            .collect();
        matched.retain(|e| !neg_entities.contains(e));
    }

    matched.sort();
    matched.dedup();
    matched
}

// ── Minigraf evaluator ────────────────────────────────────────────────────────

/// Loads `facts` into an in-memory Minigraf and runs `query`.
/// Returns a sorted list of matching entity indices (as strings, then parsed).
fn minigraf_eval(facts: &[TestFact], query: &TestQuery) -> Vec<usize> {
    let db = Minigraf::in_memory().unwrap();

    // Transact all facts.
    for fact in facts {
        let entity_kw = format!(":e{}", fact.entity);
        let val_edn = fact.value.to_edn();
        let attr = &fact.attribute;
        let edn = format!(r#"(transact [[{entity_kw} {attr} {val_edn}]])"#);
        let _ = db.execute(&edn); // ignore errors from duplicate facts
    }

    // Build the Datalog query.
    let val_clause = match &query.value_filter {
        Some(v) => format!(" (= ?v {})", v.to_edn()),
        None => String::new(),
    };

    let neg_clause = match &query.negation_attr {
        Some(neg) => format!(" (not [?e {} _])", neg),
        None => String::new(),
    };

    let attr = &query.attribute;
    let datalog = format!(
        "(query [:find ?e :where [?e {attr} ?v]{val_clause}{neg_clause}])"
    );

    match db.execute(&datalog) {
        Ok(QueryResult::QueryResults { results, .. }) => {
            let mut entities: Vec<usize> = results
                .into_iter()
                .flat_map(|r| r.into_values())
                .filter_map(|v| match v {
                    minigraf::graph::types::Value::Keyword(kw) => {
                        kw.strip_prefix(":e").and_then(|s| s.parse().ok())
                    }
                    _ => None,
                })
                .collect();
            entities.sort();
            entities.dedup();
            entities
        }
        Err(_) => vec![],
    }
}

// ── proptest generators ───────────────────────────────────────────────────────

fn arb_attribute() -> impl Strategy<Value = String> {
    prop_oneof![
        Just(":color".to_string()),
        Just(":size".to_string()),
        Just(":active".to_string()),
        Just(":tag".to_string()),
        Just(":score".to_string()),
    ]
}

fn arb_value() -> impl Strategy<Value = TestValue> {
    prop_oneof![
        Just(TestValue::Str("red".to_string())),
        Just(TestValue::Str("blue".to_string())),
        Just(TestValue::Int(1)),
        Just(TestValue::Int(2)),
        Just(TestValue::Bool(true)),
    ]
}

fn arb_fact(max_entity: usize) -> impl Strategy<Value = TestFact> {
    (0..max_entity, arb_attribute(), arb_value()).prop_map(|(entity, attribute, value)| {
        TestFact { entity, attribute, value }
    })
}

fn arb_small_graph() -> impl Strategy<Value = Vec<TestFact>> {
    prop::collection::vec(arb_fact(8), 3..20)
}

fn arb_query(facts: &[TestFact]) -> impl Strategy<Value = TestQuery> {
    // Extract the unique attributes present in the graph for realistic queries.
    let attrs: Vec<String> = {
        let mut v: Vec<String> = facts.iter().map(|f| f.attribute.clone()).collect();
        v.sort();
        v.dedup();
        if v.is_empty() { vec![":color".to_string()] } else { v }
    };
    let neg_attrs = attrs.clone();

    let attr_strategy = prop::sample::select(attrs);
    let neg_attr_strategy = prop::sample::select(neg_attrs);

    (attr_strategy, any::<bool>(), arb_value(), any::<bool>(), neg_attr_strategy).prop_map(
        |(attribute, use_value, value, use_neg, neg_attr)| TestQuery {
            attribute: attribute.clone(),
            value_filter: if use_value { Some(value) } else { None },
            negation_attr: if use_neg && neg_attr != attribute {
                Some(neg_attr)
            } else {
                None
            },
        },
    )
}

// ── Property tests ────────────────────────────────────────────────────────────

proptest! {
    /// Minigraf results must match the reference evaluator for random small graphs.
    #[test]
    fn query_results_match_reference(facts in arb_small_graph()) {
        let query_strategy = arb_query(&facts);
        // Use a fixed seed for the inner query generation to keep the test deterministic.
        let mut runner = proptest::test_runner::TestRunner::default();
        let query = query_strategy.new_tree(&mut runner).unwrap().current();

        let ref_result = ref_eval(&facts, &query);
        let mg_result = minigraf_eval(&facts, &query);

        prop_assert_eq!(
            ref_result,
            mg_result,
            "reference and minigraf disagree for query={:?}",
            query
        );
    }

    /// Negation: entities with negation_attr must not appear in results.
    #[test]
    fn negation_excludes_correct_entities(facts in arb_small_graph()) {
        // Always use negation for this test.
        let attrs: Vec<String> = {
            let mut v: Vec<String> = facts.iter().map(|f| f.attribute.clone()).collect();
            v.sort();
            v.dedup();
            v
        };
        prop_assume!(attrs.len() >= 2); // need at least 2 attributes for negation

        let query = TestQuery {
            attribute: attrs[0].clone(),
            value_filter: None,
            negation_attr: Some(attrs[1].clone()),
        };

        let ref_result = ref_eval(&facts, &query);
        let mg_result = minigraf_eval(&facts, &query);

        // Negation check: no entity in the result should also have the negation attr.
        let neg_entities: std::collections::HashSet<usize> = facts
            .iter()
            .filter(|f| f.attribute == attrs[1])
            .map(|f| f.entity)
            .collect();

        for &e in &mg_result {
            prop_assert!(
                !neg_entities.contains(&e),
                "entity {e} has negated attribute but appears in result"
            );
        }

        prop_assert_eq!(ref_result, mg_result);
    }

    /// Impossible paths return empty results, not broad catch-all output.
    #[test]
    fn impossible_path_returns_empty(facts in arb_small_graph()) {
        // Query for an attribute that does not exist in the graph.
        let query = TestQuery {
            attribute: ":__nonexistent_attr__".to_string(),
            value_filter: None,
            negation_attr: None,
        };

        let ref_result = ref_eval(&facts, &query);
        let mg_result = minigraf_eval(&facts, &query);

        prop_assert!(ref_result.is_empty(), "reference: impossible path must be empty");
        prop_assert!(mg_result.is_empty(), "minigraf: impossible path must be empty");
    }
}
```

- [ ] **Step 3: Create .github/workflows/coverage-gates.yml**

```yaml
name: Coverage Gates

on:
  push:
    branches: ["main"]
  pull_request:
    branches: ["main"]

permissions:
  contents: read

jobs:
  coverage-gates:
    name: Per-module coverage gates
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v4

      - name: Install Rust stable
        uses: dtolnay/rust-toolchain@stable
        with:
          components: llvm-tools-preview

      - name: Install cargo-llvm-cov
        uses: taiki-e/install-action@cargo-llvm-cov

      - name: Generate coverage report (JSON)
        run: cargo llvm-cov --json --output-path coverage.json

      - name: Check per-module thresholds
        run: |
          python3 - <<'PYEOF'
          import json, sys, tomllib

          with open("coverage.json") as f:
              data = json.load(f)

          with open("coverage-thresholds.toml", "rb") as f:
              thresholds = tomllib.load(f)["modules"]

          failed = []
          file_data = {}
          for entry in data.get("data", [{}])[0].get("files", []):
              fn = entry.get("filename", "")
              # Strip leading path components to match threshold keys.
              for key in thresholds:
                  if fn.endswith(key.lstrip("/")):
                      pct = entry["summary"]["lines"]["percent"]
                      file_data[key] = pct
                      if pct < thresholds[key]:
                          failed.append(f"  {key}: {pct:.1f}% < {thresholds[key]}% required")

          for key in thresholds:
              if key not in file_data:
                  print(f"WARNING: {key} not found in coverage report — skipping")

          if failed:
              print("Coverage gate FAILED:")
              for line in failed:
                  print(line)
              sys.exit(1)
          else:
              print("All coverage gates passed.")
              for k, v in sorted(file_data.items()):
                  print(f"  {k}: {v:.1f}% >= {thresholds[k]}%")
          PYEOF
```

Note: `tomllib` requires Python 3.11+. GitHub Actions ubuntu-latest has Python 3.11+. If the runner uses an older Python, replace `tomllib` with `tomli` (install via `pip install tomli`) or parse the TOML with a simpler grep-based approach.

Alternative grep-based threshold check script (if Python TOML is unavailable):

```bash
# Simpler alternative: use cargo llvm-cov --text and grep
cargo llvm-cov --text 2>&1 > cov_text.txt
check_module() {
    local file=$1 threshold=$2
    local pct=$(grep "$file" cov_text.txt | grep -oP '\d+\.\d+(?=%)' | tail -1)
    if [ -z "$pct" ]; then echo "WARNING: $file not found"; return 0; fi
    python3 -c "import sys; sys.exit(0 if float('$pct') >= $threshold else 1)" \
        || { echo "FAIL: $file: ${pct}% < ${threshold}%"; return 1; }
    echo "OK: $file: ${pct}% >= ${threshold}%"
}
FAIL=0
check_module "wal.rs"            80 || FAIL=1
check_module "db.rs"             75 || FAIL=1
check_module "persistent_facts"  75 || FAIL=1
check_module "btree_v6"          70 || FAIL=1
check_module "executor.rs"       80 || FAIL=1
check_module "evaluator.rs"      80 || FAIL=1
check_module "stratification"    80 || FAIL=1
exit $FAIL
```

Choose whichever approach works reliably on ubuntu-latest. The grep-based version is more robust.
